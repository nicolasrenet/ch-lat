Tips:

Searching FSFB by collection name:

	cd COLLECTIONS; for i in $(echo */CO.url.txt); do found=$(grep -i BaumOCist $i); test -n "$found" && echo $i ; done

Searching a FSDB collection for a particular charter:

	for i in $(echo */CH.url.txt); do found=$(grep -i 1335_III_21 $i); test -n "$found" && echo $i ; done

Searching a FSDB collection for charters with tenor:

	for cei in $(echo */CH.cei.xml) ; do found=$(grep '<cei:tenor>' $cei) ; test -n "$found" && echo $cei ; done/

Search all charters that are not part of a 'Urkunden' collection and have a tenor:

	for coll in *; do test -f $coll/CO.url.txt || continue; urkunden=$(grep -i urkunden $coll/CO.url.txt) ; test -n "$urkunden" && continue; echo $coll ; for charter in $coll/*/CH.cei.xml; do urkund=(grep 'Urkunden' $charter); test -z "$urkund" || continue; tenor=$(grep '<cei:tenor>' $charter); test -n "$tenor" && echo -e "\t$charter" ; done   ; done

Systematic search in all non-collection folders:

	(for cei in  $(find . -name "CH.cei.xml"|grep -v 'COLLECTIONS') ; do has_img=$(find $(dirname $cei) -name "*.img.jpg"); test -z "$has_img" && continue ; test -z "$(grep '<cei:tenor>' $cei)" && continue ; echo $cei ; done) > ~/tmp/data/fsdb_work/more_full_text_candidates.txt


Rsync full-text charters on local disk:

cd fsdb_full_text
for cei in $(cat ../more_full_text_candidates.txt ); do dir=$(dirname $cei | sed 's/\/mnt\/nas\/fsdb\///') ; echo $dir ; mkdir -p $dir ; rsync -a $(dirname $cei)/ $dir/  ; done
